# 查看文件上传报文

准备两个东西：

1. 前端上传文件的网页
2. 后端匹配、处理上传文件请求的规则

```ejs
<!--views/portrait.ejs-->
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>文件上传</title>
</head>
<body>
    <h2>文件上传</h2>
    <!-- enctype="multipart/form-data" 是文件上传的一个必要属性，少了它文件就传不上去 -->
    <!-- multi 许多 part 部分 form 表单 data 数据-->
    <!-- 上传文件如何不设置这个属性，也能提交，但请求报文的上传文件部分只会包含文件名，请求体中不包含文件内容-->
    <form action="/portrait" method='post' enctype="multipart/form-data">
        用户名：<input type="text" name="username"><br>
        头像：<input type="file" name="portrait"><br>
        <hr>
        <button>点击提交</button>
    </form>
</body>
</html>
```

```js
// routes/index.js
var express = require('express');
var router = express.Router();

/* GET home page. */
router.get('/', function(req, res, next) {
  res.render('index', { title: 'Express' });
});

// 显示网页的表单 portrait是肖像的意思
router.get('/portrait', (req, res) => {
  res.render("portrait");
})
// 处理文件上传
router.post('/portrait', (req, res) => {
  res.send("成功");
})

module.exports = router;
```
